    <script>
        var allCustomerData = JSON.parse(localStorage.getItem('electricityHistory')) || [];
        updateTable();

        function generateBill() {
            var name = document.getElementById('userName').value || "คุณลูกค้า";
            var prev = parseFloat(document.getElementById('prevUnit').value) || 0;
            var curr = parseFloat(document.getElementById('currUnit').value);
            var rate = parseFloat(document.getElementById('rate').value) || 0;

            if (isNaN(curr) || curr < prev) {
                alert("กรุณากรอกข้อมูลให้ถูกต้อง");
                return;
            }

            var units = curr - prev;
            var total = units * rate;

            var newData = { id: Date.now(), name: name, prev: prev, curr: curr, units: units, rate: rate, total: total };
            allCustomerData.push(newData);
            localStorage.setItem('electricityHistory', JSON.stringify(allCustomerData));
            
            showReceipt(newData);
            updateTable();
        }

        function showReceipt(data) {
            document.getElementById('resName').innerText = data.name;
            document.getElementById('resPrev').innerText = data.prev.toLocaleString();
            document.getElementById('resCurr').innerText = data.curr.toLocaleString();
            document.getElementById('resUnitUsed').innerText = data.units.toLocaleString() + " หน่วย";
            document.getElementById('resRate').innerText = data.rate.toFixed(2);
            document.getElementById('resGrand').innerText = data.total.toLocaleString(undefined, {minimumFractionDigits: 2});

            document.getElementById('receipt-area').style.display = 'block';
            document.getElementById('actionArea').style.display = 'flex';
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function updateTable() {
            const body = document.getElementById('historyBody');
            body.innerHTML = "";
            allCustomerData.forEach((item, index) => {
                const row = `<tr>
                    <td>${index + 1}</td>
                    <td class="clickable-name" onclick="viewOldBill(${item.id})">${item.name}</td>
                    <td>${item.units}</td>
                    <td>฿${item.total.toLocaleString()}</td>
                </tr>`;
                body.innerHTML += row;
            });
        }

        function viewOldBill(id) {
            const data = allCustomerData.find(item => item.id === id);
            if(data) showReceipt(data);
        }

        function downloadImage() {
            // ถ่ายรูปเฉพาะส่วน #receipt-area เท่านั้น (ซึ่งไม่มีปุ่มดาวน์โหลดแล้ว)
            const area = document.getElementById('receipt-area');
            html2canvas(area, { 
                scale: 3, 
                backgroundColor: "#dfe6e9" 
            }).then(canvas => {
                var link = document.createElement('a');
                link.download = 'Invoice_6942.png';
                link.href = canvas.toDataURL();
                link.click();
            });
        }

        function exportToExcel() {
            if (allCustomerData.length === 0) return;
            const exportData = allCustomerData.map(({id, ...rest}) => ({
                "ชื่อลูกค้า": rest.name, "เลขมิเตอร์ก่อน": rest.prev, "เลขมิเตอร์หลัง": rest.curr,
                "หน่วยที่ใช้": rest.units, "ราคา/หน่วย": rest.rate, "ยอดเงินสุทธิ": rest.total
            }));
            var ws = XLSX.utils.json_to_sheet(exportData);
            var wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "Data");
            XLSX.writeFile(wb, "ข้อมูลค่าไฟ.xlsx");
        }
    </script>
